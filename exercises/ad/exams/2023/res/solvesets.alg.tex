\begin{algorithm}[H]
  \caption{\textalgo{solvesets($M$, $L$)}}\label{alg:solvesets}

  \KwData{A list $M$ of $n$ integers and an integer $L$}
  \KwResult{An integer $l$}

  \lIf{$n = 1$}{\Return{$L[M[0]]$}}
  \Else{
    $l_1 \gets$ \textalgo{solvesets($M[:n-1]$, $L$)}\;
    $l_2 \gets$ \textalgo{solvesets($M[n:]$, $L$)}\;
    \Return{\textalgo{Alg($M, L\cup\{l_1, l_2\}$)}}\;
  }
\end{algorithm}
