\begin{alg}
  \signed{dynamicperrin}{n}{result}
  \params{$n \in \mathbb{N}$}{$\text{result} \in \mathbb{N}$}

  $n \gets [3,0,2]$\;

  \For{$i \gets 0$ \KwTo $n$}{
    $n[3] \gets n[0] + n[1]$\;
    $n[0] \gets n[1]$\;
    $n[1] \gets n[2]$\;
    $n[2] \gets n[3]$\;
  }

  \Return{$n_{-1}$}
\end{alg}
