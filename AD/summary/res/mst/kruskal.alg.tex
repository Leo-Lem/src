\begin{alg}
  \signed{kruskal}{G}{T}
  \params{A graph $G$}{A minimal spanning tree $T$}
  \BlankLine

  $T \gets \emptyset$\;
  \lForEach{vertex $v$ of $G$}{
    \texttt{makeSet($v$)}
  }
  sort edges of $G$ by weight\;
  \ForEach{edge $(u, v)$ of $G$}{
    \If{\texttt{findSet($u$)} $\neq$ \texttt{findSet($v$)}}{
      $T \gets T \cup \{(u, v)\}$\;
      \texttt{union($u, v$)}
    }
  }
  \Return $T$\;
\end{alg}
