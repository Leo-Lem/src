\begin{algorithm}[H]
  \caption{\textalgo{WhoAmI(A)}}\label{alg:whoami}
  \KwData{Array $A$ von $n \geq 1$ natürlichen Zahlen}
  \BlankLine

  $x \gets 1$\;
  $m \gets 1$\;
  \For{$i \gets 2$ \KwTo $n$} {
    \If{$A[i - 1] \leq A[i]$} {
      $m \gets 0$\;
    }
    $m \gets m + 1$\;
    $x \gets \max(x, m)$\;
  }
  \Return{$x$}\;
\end{algorithm}
