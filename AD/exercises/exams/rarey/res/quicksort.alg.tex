\begin{alg}
  \signed{Partition}{A, l, r}{\empty}
  \params{$l$ - linkes Blockende, $r$ - rechtes Blockende, $A$ - Array}{\empty}
  $i \gets l - 1$\;
  \For{$j \gets l$ \KwTo $r - 1$}{
    \If{$A[j] \leq A[r]$}{
      $i \gets i + 1$\;
      swap $A[i], A[j]$\;
    }
  }
  swap $A[i + 1], A[r]$\;
  \KwRet{$(i + 1)$}\;
\end{alg}
