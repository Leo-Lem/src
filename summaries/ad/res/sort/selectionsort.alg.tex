\begin{alg}
  \signed{selectionSort}{A}{A_\leq}
  \params{An array $A$ of $n$ elements}{The array $A$ sorted in non-decreasing order}

  \For{$i \gets 1$ \KwTo $n-1$}{
    $min \gets i$\;
    \For{$j \gets i+1$ \KwTo $n$}{
      \If{$A[j] < A[min]$}{
        $min \gets j$\;
      }
    }
    swap $A[i]$ and $A[min]$\;
  }
\end{alg}
