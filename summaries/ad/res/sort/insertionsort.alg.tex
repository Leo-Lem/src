\begin{alg}
  \signed{insertionSort}{A}{A_{\leq}}
  \params{An array $A$ of $n$ elements}{The array $A$ sorted in non-decreasing order}

  \For{$i \gets 2$ \KwTo $n$}{
    $key \gets A[i]$\;
    $j \gets i-1$\;
    \While{$j > 0$ \textbf{and} $A[j] > key$}{
      $A[j+1] \gets A[j]$\;
      $j \gets j-1$\;
    }
    $A[j+1] \gets key$\;
  }
\end{alg}
