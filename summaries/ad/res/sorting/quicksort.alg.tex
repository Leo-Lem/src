\begin{alg}
  \signed{quickSort}{A, p, r}{A_\leq}
  \params{An array $A$ of $n$ elements, with indices $p$ and $r$}{The array $A$ sorted in non-decreasing order}

  \If{$p < r$}{
    $q \gets$ \textalgo{partition($A$, $p$, $r$)}\;
    \textalgo{quickSort($A$, $p$, $q-1$)}\;
    \textalgo{quickSort($A$, $q+1$, $r$)}\;
  }
\end{alg}
