\begin{alg}
  \signed{heapSort}{A}{A_{\leq}}
  \params{An array $A$ of $n$ elements}{The array $A$ sorted in non-decreasing order}

  \For{$i \gets \lfloor\frac{n}{2}\rfloor$ \KwTo $1$}{
    \textalgo{heapify($A$, $i$, $n$)}\;
  }
  \For{$i \gets n$ \KwTo $2$}{
    swap $A[1]$ and $A[i]$\;
    \textalgo{heapify($A$, $1$, $i-1$)}\;
  }
\end{alg}
