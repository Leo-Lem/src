\begin{algorithm}[H]
  \caption{\textalgo{selectionSort($A$)}}\label{alg:selectionsort}

  \KwData{An array $A$ of $n$ elements}
  \KwResult{The array $A$ sorted in non-decreasing order}
  \BlankLine

  \For{$i \gets 1$ \KwTo $n-1$}{
    $min \gets i$\;
    \For{$j \gets i+1$ \KwTo $n$}{
      \If{$A[j] < A[min]$}{
        $min \gets j$\;
      }
    }
    swap $A[i]$ and $A[min]$\;
  }
\end{algorithm}
