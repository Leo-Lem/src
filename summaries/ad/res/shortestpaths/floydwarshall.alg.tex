\begin{alg}
  \signed{floydWarshall}{W}{D}
  \params{Weighted adjacency matrix $W$ of a graph $G$}{Matrix of shortest distances $D$}

  $n \gets \text{number of vertices in } G$\;
  $D \gets W$\;

  \For{$k \gets 1$ \KwTo $n$}{
    \For{$i \gets 1$ \KwTo $n$}{
      \For{$j \gets 1$ \KwTo $n$}{
        $D[i,j] \gets \min(D[i,j], D[i,k] + D[k,j])$
      }
    }
  }

  \Return $D$\;
\end{alg}
